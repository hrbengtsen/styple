### Examples

<Preview>
  <CommandPaletteContainer />
</Preview>

```tsx outline=true
import { CommandPalette } from "@styple/design-system";

const dummyData = [
  { label: "Docs page 1", isDiv: false },
  { label: "Docs page 2", isDiv: false },
  { label: "Docs page 3", isDiv: false },
  { label: "Final pages", isDiv: true },
  { label: "Last docs page", isDiv: false },
];

// Boilerplate component of how to implement a CommandPalette
export const CommandPaletteContainer = () => {
  const [query, setQuery] = useState("");

  const filteredData = query === "" ? dummyData : dummyData.filter((item) => (item.toLowerCase().includes(query.toLowerCase())));

  return (
    <CommandPalette
      query={query}
      setQuery={setQuery} 
      data={filteredData}
    />
  );
};
```
